//selection sort
class Solution {
    public int[] selectionSort(int[] nums) {
        int minindex=0;
        for(int i=0;i<nums.length;i++)
        {
            minindex=i;
            for(int j=i+1;j<nums.length;j++)
            {
                if(nums[j]<nums[minindex])
                minindex=j;
            }
            int temp=nums[minindex];
            nums[minindex]=nums[i];
            nums[i]=temp;
        }
        return nums;
    }
}
