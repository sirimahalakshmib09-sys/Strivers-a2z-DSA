//Remove Duplicates from sorted array
//Brute force approach
//Time Complexity-O(N)
//Space Complexity-O(N)
class Solution {
    public int removeDuplicates(int[] nums) {
        HashSet<Integer> unique=new HashSet<>();
        int index=0;
        for(int num:nums)
        {
            if(!unique.contains(num))
            {
            unique.add(num);
            nums[index]=num;
            index++;
            }
        }
        return index;
    }
}
